暑假在云南村子里实习的时候写了一个文本游戏的稿子，回来的时候就想拿一些在那边拍的照片做成一个视觉小说（不会画画只能用摄影凑合：）），也想看看能不能把摄影的美学风格和视觉小说做一个结合。一个主要的灵感来源是Chris Marker的《堤》，我想在unity里复刻这种胶片的感觉。

文字的部分我则想做成类似《极乐迪斯科》里的对话栏，而不是renpy那种每次只呈现一句的形式，这样玩家能看到自己之前的选项，而且这篇稿子也不以角色之间的对话为主，和常规的视觉小说有些不同。

文字部分并不是很难实现，下一篇博客再讲。博主之前没有自己写过shader（再次感叹学校的CG课真是一点实用的都不教…），所以就记录一下这个学习过程。

首先去The Book of Shaders了解了一下glsl和shader的工作流。但这本书的作者是一个数字艺术家，所以有很多内容是在讲有用的生成性算法和函数。这个以后估计会用到。

看了第一部分之后就直接去Shader Toy搜索了我想要的效果（胶片颗粒，Film Grain）然后边看大神代码边学。找到了两个效果比较不错的：Weasletron、terchapone

先来看这个比较复杂的:

代码 
const float permTexUnit = 1.0/256.0;		// Perm texture texel-size
const float permTexUnitHalf = 0.5/256.0;	// Half perm texture texel-size
   
#define width float(iResolution.x)
#define height float(iResolution.y)

const float grainamount = 0.01; //grain amount
bool colored = false; //colored noise?
float coloramount = 0.6;
float grainsize = 1.5; //grain particle size (1.5 - 2.5)
float lumamount = 2.0; //

//a random texture generator, but you can also use a pre-computed perturbation texture
vec4 rnm(in vec2 tc) 
{
    float timer = iTime;
    float noise =  sin(dot(tc + vec2(timer,timer),vec2(12.9898,78.233))) * 43758.5453;

    float noiseR =  fract(noise)*2.0-1.0;
    float noiseG =  fract(noise*1.2154)*2.0-1.0; 
    float noiseB =  fract(noise*1.3453)*2.0-1.0;
    float noiseA =  fract(noise*1.3647)*2.0-1.0;
    
    return vec4(noiseR,noiseG,noiseB,noiseA);
}

float fade(in float t) {
    return t*t*t*(t*(t*6.0-15.0)+10.0);
}

float pnoise3D(in vec3 p)
{
    vec3 pi = permTexUnit*floor(p)+permTexUnitHalf; // Integer part, scaled so +1 moves permTexUnit texel
    // and offset 1/2 texel to sample texel centers
    vec3 pf = fract(p);     // Fractional part for interpolation

    // Noise contributions from (x=0, y=0), z=0 and z=1
    float perm00 = rnm(pi.xy).a ;
    vec3  grad000 = rnm(vec2(perm00, pi.z)).rgb * 4.0 - 1.0;
    float n000 = dot(grad000, pf);
    vec3  grad001 = rnm(vec2(perm00, pi.z + permTexUnit)).rgb * 4.0 - 1.0;
    float n001 = dot(grad001, pf - vec3(0.0, 0.0, 1.0));

    // Noise contributions from (x=0, y=1), z=0 and z=1
    float perm01 = rnm(pi.xy + vec2(0.0, permTexUnit)).a ;
    vec3  grad010 = rnm(vec2(perm01, pi.z)).rgb * 4.0 - 1.0;
    float n010 = dot(grad010, pf - vec3(0.0, 1.0, 0.0));
    vec3  grad011 = rnm(vec2(perm01, pi.z + permTexUnit)).rgb * 4.0 - 1.0;
    float n011 = dot(grad011, pf - vec3(0.0, 1.0, 1.0));

    // Noise contributions from (x=1, y=0), z=0 and z=1
    float perm10 = rnm(pi.xy + vec2(permTexUnit, 0.0)).a ;
    vec3  grad100 = rnm(vec2(perm10, pi.z)).rgb * 4.0 - 1.0;
    float n100 = dot(grad100, pf - vec3(1.0, 0.0, 0.0));
    vec3  grad101 = rnm(vec2(perm10, pi.z + permTexUnit)).rgb * 4.0 - 1.0;
    float n101 = dot(grad101, pf - vec3(1.0, 0.0, 1.0));

    // Noise contributions from (x=1, y=1), z=0 and z=1
    float perm11 = rnm(pi.xy + vec2(permTexUnit, permTexUnit)).a ;
    vec3  grad110 = rnm(vec2(perm11, pi.z)).rgb * 4.0 - 1.0;
    float n110 = dot(grad110, pf - vec3(1.0, 1.0, 0.0));
    vec3  grad111 = rnm(vec2(perm11, pi.z + permTexUnit)).rgb * 4.0 - 1.0;
    float n111 = dot(grad111, pf - vec3(1.0, 1.0, 1.0));

    // Blend contributions along x
    vec4 n_x = mix(vec4(n000, n001, n010, n011), vec4(n100, n101, n110, n111), fade(pf.x));

    // Blend contributions along y
    vec2 n_xy = mix(n_x.xy, n_x.zw, fade(pf.y));

    // Blend contributions along z
    float n_xyz = mix(n_xy.x, n_xy.y, fade(pf.z));

    // We're done, return the final noise value.
    return n_xyz;
}

//2d coordinate orientation thing
vec2 coordRot(in vec2 tc, in float angle)
{
    float aspect = width/height;
    vec2 uv = (tc*2.0)-1.0;
    float rotX = (uv.x*aspect*cos(angle)) - (uv.y*sin(angle));
    float rotY = (uv.y*cos(angle)) + (uv.x*aspect*sin(angle));
//	float rotX = ((tc.x*2.0-1.0)*aspect*cos(angle)) - ((tc.y*2.0-1.0)*sin(angle));
//	float rotY = ((tc.y*2.0-1.0)*cos(angle)) + ((tc.x*2.0-1.0)*aspect*sin(angle));
    rotX = ((rotX/aspect)*0.5+0.5);
    rotY = rotY*0.5+0.5;
    return vec2(rotX,rotY);
}

void mainImage( out vec4 fragColor, in vec2 fragCoord ) 
{
    float timer = iTime;
    vec2 texCoord = fragCoord.rg / vec2(iResolution.x, iResolution.y);
    
    vec3 rotOffset = vec3(1.425,3.892,5.835); //rotation offset values	
    vec2 rotCoordsR = coordRot(texCoord, timer + rotOffset.x);
    vec3 noise = vec3(pnoise3D(vec3(rotCoordsR*vec2(width/grainsize,height/grainsize),0.0)));
  
    if (colored)
    {
        vec2 rotCoordsG = coordRot(texCoord, timer + rotOffset.y);
        vec2 rotCoordsB = coordRot(texCoord, timer + rotOffset.z);
        noise.g = mix(noise.r,pnoise3D(vec3(rotCoordsG*vec2(width/grainsize,height/grainsize),1.0)),coloramount);
        noise.b = mix(noise.r,pnoise3D(vec3(rotCoordsB*vec2(width/grainsize,height/grainsize),2.0)),coloramount);
    }

    vec3 col = texture(iChannel0, texCoord).rgb;

    //noisiness response curve based on scene luminance
    vec3 lumcoeff = vec3(0.299,0.587,0.114);
    float luminance = lumamount * dot(col, lumcoeff);
    float lum = smoothstep(0.2,0.0,luminance);
    lum += luminance;
    
    noise = mix(noise,vec3(0.0),pow(lum,4.0));
    col = col-noise*grainamount;
    
    vec3 grayColor = vec3(0.299*col.r + 0.587*col.g + 0.114*col.b);
   
    fragColor =  vec4(col,1.0);
    
}

众所周知Shader（着色器）就是交给GPU并行运行的图像处理程序，每个线程都用这个程序渲染整个图像的一小部分。着色器一般有两部分，顶点着色器（Vertex Shader）和片元着色器（Fragment Shader）。顾名思义，顶点着色器负责“加工”和顶点相关的数据，比如《风之旅人》就在顶点法线上做了一些手脚，让游戏里的沙子在光线的作用下更加漂亮。片元着色器则根据顶点数据拾取文理并输出颜色。因为我只是想给现成的摄影图片在runtime加工，所以并不需要顶点着色器，只用片元就行。这几个ShaderToy的代码也都只有片元的部分（mainImage）。

最上面是两个辅助函数rnm和fade。Fade相当于一个smoothstep阶跃函数（6t^5-15t^4+10t^3）。rnm则是一个随机数生成器：

vec4 rnm(in vec2 tc) 
{
    float timer = iTime;
    float noise =  sin(dot(tc + vec2(timer,timer),vec2(12.9898,78.233))) * 43758.5453;

    float noiseR =  fract(noise)*2.0-1.0;
    float noiseG =  fract(noise*1.2154)*2.0-1.0; 
    float noiseB =  fract(noise*1.3453)*2.0-1.0;
    float noiseA =  fract(noise*1.3647)*2.0-1.0;
    
    return vec4(noiseR,noiseG,noiseB,noiseA);
}

它把输入的纹理坐标加上当前的程序时间（每次都生成新的噪音纹理来达成影片中不断变换的的噪点效果），与一个设定好的向量做点积，丢进sin里，再取结果的的小数部分（一个经典的随机数生成方法）。当前的程序时间和纹理坐标就相当于随机种子。最后，pNoise3D用这个函数用Simplex Noise算法（和Perlin Noise类似的一种具有连贯性的噪音算法）生成我们需要的噪音纹理。

coordRot则是一个用于旋转纹理坐标的辅助函数，还是用来营造电影镜头里那种不断变换的颗粒感。经过旋转过后的纹理坐标coordRot与vec2(width/grainsize,height/grainsize)（对应元素）相乘之后丢进pNoise里生成这个坐标的噪点数据。grainsize顾名思义可以用来调试颗粒大小，因为越大的grainsize意味着每个纹理坐标之间覆盖的噪音纹理区域更小了（相当于变相把噪音的像素降低了）。

最后的最后的最后，程序还通过测定该片元的亮度来调试噪点的程度。lumcoeff代表的是人眼观测不同颜色的天生比重（在相同的“亮度”下，人会角色绿色更“亮”）。亮度越高的片元噪点会更大。把噪点值从原有的纹理颜色值减去之后就大功告成了！

理解这个再理解第二个比较简单的就容易多了。它直接调用了一个类似rnm的随机噪音。这个没有随时间变换也没有根据亮度调试。

void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    // Normalized pixel coordinates (from 0 to 1)
    vec2 uv = fragCoord/iResolution.xy;

    // Calculate noise and sample texture
    float mdf = 0.1; // increase for noise amount 
    float noise = (fract(sin(dot(uv, vec2(12.9898,78.233)*2.0)) * 43758.5453));
    vec4 tex = texture(iChannel0, uv);
    
    mdf *= sin(iTime) + 1.0; // animate the effect's strength
    
    vec4 col = tex - noise * mdf;

    // Output to screen
    fragColor = col;
}

处理完颗粒感来做相机抖动。《堤》不知道是如何拍摄的，可以看到每一张照片都有小幅度的抖动。因为镜头再怎样抖动都是有连贯性的，且会自己向中心调整，所以我们还是用噪音函数来处理：

public class ImageMovement : MonoBehaviour
{
    Vector3 init;
    public float mag = 1;
    public float sharpness = 3;
    public float perlinWeight = 0.9f;

    // Start is called before the first frame update
    void Start()
    {
        init = transform.position;
    }

    // Update is called once per frame
    void Update()
    {
        if (Input.GetKey(KeyCode.E))
        {
            transform.position = init + perlinWeight * PerlinMove(init, mag) + (1-perlinWeight) * RandomMove2(init, mag);
        }
    }

    public float norm(float x)
    {
        return 2 * x - 1;
    }

    // Use sharpness multiplier to downsample and increase jitters
    public Vector3 PerlinMove(Vector3 init, float mag)
    {
        return mag * new Vector3(norm(Mathf.PerlinNoise(Time.time*sharpness, 0)), norm(Mathf.PerlinNoise(0, Time.time*sharpness)), 0);
    }

    public Vector3 RandomMove2(Vector3 init, float mag)
    {
        return mag * new Vector3(norm((Random.value + Random.value + Random.value) / 3), norm((Random.value + Random.value + Random.value) / 3), 0);
    }
}
